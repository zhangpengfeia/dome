## 获取jq对象的尺寸和位置

```js
jq.width() //直接获取宽度，不带px,只能获取
jq.height()

innerWidth() //w+p
innerHieght() 

outerWidth() // w+p+b
outerHieght()

outerWidth(true) //w+p+b+m
outerHeight(true) 

jq.prsition(); //获取位置，返回的是位置对象，只能获取


jq.offset({
	top:1
    left:2
}) //设置文档距离，根据左上角计算



jq.scroll(function(){
    jq.scrollTop()//获取卷入高度
    jq.scrollLeft(200) //设置左侧滚动条
}) //滚动事件



//带动画的返回,只能设置给body和html
$('body,html').stop().animate(){
    sorcollTop:0
}



```
